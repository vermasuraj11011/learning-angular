# attribute and class selector

@Component({
  // selector: 'app-server',   -> mostly used way
  // selector :  '[app-server]', -> component as attribute
  // selector : '.app-server' -> component as class
  templateUrl: './server.component.html'
})


# property binding
[property name]='<attribute name>'
eg:
[disable]='isServerDisable'

# event binding
(event name)='<event handler>'
(click)='disableTheServer()'

# two way binding
[(ngModel)]='serverName'

# directive
ngIf, ngFor, ngStyle, ngClass
